137  void animate()
138  {
139  if (_is_animated == false)
140  return;
141  if (this->_curr_frame > this->_nb_frame)
142  {
143  this->_texRect.top = 0;
144  this->_texRect.left = 0;
145  this->_curr_frame = 0;
146  }
147 
148  int currWidth = (this->_texRect.left + this->_texRect.width);
149  int currHeight = (this->_texRect.top + this->_texRect.height);
150  int maxWidth = this->_dimension.x;
151  int maxHeight = this->_dimension.y;
152 
153  this->_texRect.left += this->_steps.x;
154  // this->_texRect = sf::IntRect((sf::Vector2i){this->_texRect.left + this->_steps.x, this->_texRect.top}, (sf::Vector2i){this->_texRect.width, this->_texRect.height});
155  if (currWidth > maxWidth)
156  {
157  this->_texRect.top += this->_steps.y;
158  this->_texRect.left = 0;
159  }
160  if (currHeight > maxHeight)
161  this->_texRect.top = 0;
162  this->sprite.setTextureRect(this->_texRect);
163  this->_curr_frame += 1;
164  }
